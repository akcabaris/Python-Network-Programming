******* create ip header *******
 ip (variable),IP (header)
	>>> ip=IP()
	>>> ip.show()
	>>> ip.dst="10.10.10.21"

******* tcp *******
	>>> tcp=TCP()
	>>> tcp.show()
	>>> tcp.summary()
	>>> tcp.sport=5001
	>>> tcp.dport=5002
	>>> tcp.flags='SA'  (SYN+ACK)
	>>> tcp.flags='S'	(SYN)
	>>> tcp.flags='A'	(ACK)

******* create package *******
	>>> pckt=ip/tcp
	>>> pckt.show()
	or
	>>> pckt2=IP(dst='10.10.10.21')/TCP(sport=5001,dport=5002,flags='SA')
	>>> pckt2.show()


******* ping request *******
	>>> icmp=ICMP()
	>>> icmp.show()
	>>> ipPing=IP()
	>>> ipPing.dst='10.10.10.21'
	>>> ipPckt=ipPing/icmp
	>>> ipPckt.show()
	>>> ipPckt.summary()

******* packet sending *******
	>>> send(ipPckt)
	>>> sr(ipPckt)
	>>> rcvPckt=sr(ipPckt)
	>>> rcvPckt.summary()

	>>> ans,unans=sr(ipPckt)
	>>> ans.summary()
	>>> ans[0]
	>>> ans[0][0]
	>>> ans[0][1]
	>>> ans[0][1].show()
	>>> ans[0][0].show()

	>>> rcvPckt=sr1(ipPckt)
	>>> rcvPckt.summary()
	>>> rcvPckt.show()
